@app.route('/upload', methods=['POST'])
def upload_file():
    if 'file' not in request.files:
        return jsonify({"message": "No file part"}), 400
    file = request.files['file']
    if file.filename == '':
        return jsonify({"message": "No selected file"}), 400

    file_extension = os.path.splitext(file.filename)[1]  # Obtenha a extensão do arquivo
    allowed_extensions = {'.txt', '.csv', '.json'}  # Adicione outras extensões conforme necessário

    if file and file_extension in allowed_extensions:
        fixed_filename = 'arquivo_fixo' + file_extension  # Defina o nome fixo e preserve a extensão do arquivo
        filepath = os.path.join(app.config['UPLOAD_FOLDER'], fixed_filename)
        
        content = file.read()
        fixed_text = b"Texto Fixo: "  # Texto fixo a ser adicionado no início do arquivo
        new_content = fixed_text + content

        with open(filepath, 'wb') as f:
            f.write(new_content)

        return jsonify({"message": "File successfully uploaded", "filename": fixed_filename}), 200
    else:
        return jsonify({"message": "Invalid file type"}),
